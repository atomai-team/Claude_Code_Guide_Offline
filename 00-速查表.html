快捷技巧 + 常用组合命令

# ── 基础操作 ──────────────────────────────────
/cd ~/projects/mall         # 切目录不重开session（v2.1.169+）
/btw 这个函数叫什么名字？     # 侧问，不污染主对话
claude --safe-mode          # 纯净启动（排障用，禁所有插件/hooks）
/compact                    # 压缩长对话上下文
/rewind                     # 撤回到上一个检查点
Esc Esc                     # 中断 Claude 正在做的事

# ── 任务规划 ──────────────────────────────────
/goal "购物车页面可以正常加减商品、显示总价"   # 设任务完成条件（>5轮必用）

# ── 外部 AI 并发对抗（省Claude token）──────────
scripts/ai-panel.sh "调研 Next.js vs Nuxt.js" gemini kimi mmx
scripts/ai-cli.sh kimi "分析这份 100 页行业报告"  # 256K context

# ── 健康检查 ──────────────────────────────────
python3 ~/.claude/lib/version-consistency-check.py --json
/config-audit-better        # 配置健康审计
/audit-all-better           # 30维健康诊断
omc update                  # 更新 OMC（当前有 v4.14.6 可用）
